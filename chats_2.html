		public int next(digits, int i, int e) {
				// Find the rightmost element that is greater than nums[i]
				while (digits[e] <= digits[i]) {
					e--;
				}

				// Swap elements at i and j
				int t = digits[i];
				digits[i] = digits[j];
				digits[j] = t;
			}
			
			// Reverse the elements from i+1 to the end of the array
			reverse(nums, i + 1, e);
		}
